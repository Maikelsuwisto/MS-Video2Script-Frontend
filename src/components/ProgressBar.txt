import React from 'react'


export default function ProgressBar({ value, label }: { value: number; label?: string }) {
return (
<div style={{ width: '100%', background: '#eee', borderRadius: 8, padding: 4 }}>
<div style={{ width: `${Math.min(100, Math.max(0, value))}%`, height: 10, borderRadius: 6, background: '#4f46e5' }} />
{label && <div style={{ marginTop: 6, fontSize: 12, color: '#333' }}>{label}</div>}
</div>
)
}
